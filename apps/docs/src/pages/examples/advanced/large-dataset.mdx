# Large Dataset

10만 개 이상의 데이터 포인트를 효율적으로 처리하는 방법입니다.

## 가상화 (Virtualization)

뷰포트 밖의 데이터를 렌더링하지 않습니다:

```typescript
import {
  createChart,
  filterByViewport,
  calculateViewportFromDataset
} from '@xelstack/chart-core';

// 10만 개 데이터 포인트
const largeDataset = {
  points: Array.from({ length: 100000 }, (_, i) => ({
    x: i,
    y: Math.sin(i / 1000) * 50 + 50
  }))
};

// 초기 뷰포트 계산
const viewport = calculateViewportFromDataset(largeDataset);

// 뷰포트 내 데이터만 필터링
const visibleData = filterByViewport(largeDataset, viewport);

// 차트 생성 (필터링된 데이터만)
const chart = createChart(container, visibleData, {
  type: 'line',
  title: `Large Dataset (${largeDataset.points.length.toLocaleString()} points)`,
  showGrid: true
});
```

## 샘플링

데이터를 다운샘플링하여 성능 향상:

```typescript
import { uniformSample } from '@xelstack/chart-core';

// 균등 샘플링 - 1000개로 축소
const sampledData = uniformSample(largeDataset, 1000);

const chart = createChart(container, sampledData, {
  type: 'line',
  title: 'Sampled Data'
});
```

## 적응형 샘플링

중요한 포인트는 유지:

```typescript
import { adaptiveSample } from '@xelstack/chart-core';

// 중요도 기반 샘플링
const adaptiveSampledData = adaptiveSample(
  largeDataset,
  5000, // 목표 포인트 수
  0.7   // 중요도 가중치
);

const chart = createChart(container, adaptiveSampledData, config);
```

## 청킹 (Chunking)

데이터를 청크로 나누어 처리:

```typescript
async function loadLargeDataset(url, chunkSize = 10000) {
  const chart = createChart(container, { points: [] }, config);
  let allPoints = [];

  const response = await fetch(url);
  const reader = response.body.getReader();
  const decoder = new TextDecoder();

  let buffer = '';

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    buffer += decoder.decode(value, { stream: true });
    const lines = buffer.split('\n');
    buffer = lines.pop() || '';

    for (const line of lines) {
      if (line.trim()) {
        const [x, y] = line.split(',').map(Number);
        allPoints.push({ x, y });

        // 청크마다 차트 업데이트
        if (allPoints.length % chunkSize === 0) {
          chart.updateData({ points: [...allPoints] });
          chart.render();
          await new Promise(resolve => setTimeout(resolve, 0));
        }
      }
    }
  }

  chart.updateData({ points: allPoints });
  chart.render();
}
```

## 레이지 로딩

필요할 때만 데이터 로드:

```typescript
class LazyLoadChart {
  constructor(container, config, dataSource) {
    this.chart = createChart(container, { points: [] }, config);
    this.dataSource = dataSource;
    this.loadedRanges = [];
    this.setupViewportListener();
  }

  setupViewportListener() {
    // 뷰포트 변경 감지
    this.chart.on('viewportChange', (viewport) => {
      this.loadDataForViewport(viewport);
    });
  }

  async loadDataForViewport(viewport) {
    const range = {
      xMin: viewport.xMin,
      xMax: viewport.xMax
    };

    // 이미 로드된 범위인지 확인
    if (this.isRangeLoaded(range)) {
      return;
    }

    // 데이터 로드
    const newData = await this.dataSource.fetch(range);

    // 기존 데이터와 병합
    const currentData = this.chart.getData();
    const mergedPoints = [...currentData.points, ...newData.points]
      .sort((a, b) => a.x - b.x);

    this.chart.updateData({ points: mergedPoints });
    this.chart.render();

    this.loadedRanges.push(range);
  }

  isRangeLoaded(range) {
    return this.loadedRanges.some(loaded =>
      loaded.xMin <= range.xMin && loaded.xMax >= range.xMax
    );
  }
}
```

## Web Worker

백그라운드에서 데이터 처리:

```typescript
// worker.js
self.addEventListener('message', (e) => {
  const { data, operation } = e.data;

  if (operation === 'sample') {
    const sampled = uniformSample(data, 1000);
    self.postMessage({ result: sampled });
  } else if (operation === 'filter') {
    const filtered = data.points.filter(p => p.y > 50);
    self.postMessage({ result: { points: filtered } });
  }
});

// main.js
const worker = new Worker('worker.js');

worker.postMessage({
  operation: 'sample',
  data: largeDataset
});

worker.addEventListener('message', (e) => {
  const sampledData = e.result;
  chart.updateData(sampledData);
  chart.render();
});
```

## 진행률 표시

대용량 데이터 로딩 시:

```typescript
async function loadWithProgress(url, onProgress) {
  const response = await fetch(url);
  const contentLength = response.headers.get('Content-Length');
  const total = parseInt(contentLength, 10);

  let loaded = 0;
  const reader = response.body.getReader();
  const chunks = [];

  while (true) {
    const { done, value } = await reader.read();

    if (done) break;

    chunks.push(value);
    loaded += value.length;

    // 진행률 콜백
    onProgress({
      loaded,
      total,
      percentage: (loaded / total) * 100
    });
  }

  // 데이터 파싱
  const blob = new Blob(chunks);
  const text = await blob.text();
  const points = parseCSV(text);

  return { points };
}

// 사용
const progressBar = document.getElementById('progress');

loadWithProgress('/api/large-dataset', (progress) => {
  progressBar.style.width = `${progress.percentage}%`;
  progressBar.textContent = `${progress.percentage.toFixed(1)}%`;
}).then(dataset => {
  chart.updateData(dataset);
  chart.render();
  progressBar.style.display = 'none';
});
```

## 완전한 예제

```typescript
class LargeDatasetChart {
  constructor(container, config, options = {}) {
    this.maxVisiblePoints = options.maxVisiblePoints || 5000;
    this.useSampling = options.useSampling || true;
    this.chart = createChart(container, { points: [] }, config);
    this.fullDataset = { points: [] };
  }

  async loadData(url) {
    const progressEl = document.getElementById('loading-progress');
    progressEl.style.display = 'block';

    try {
      // 데이터 로드
      const dataset = await this.loadWithProgress(url, (progress) => {
        progressEl.textContent = `Loading... ${progress.percentage.toFixed(1)}%`;
      });

      this.fullDataset = dataset;

      // 초기 렌더링 데이터 준비
      const displayData = this.prepareDisplayData(dataset);

      this.chart.updateData(displayData);
      this.chart.render();

      progressEl.style.display = 'none';

      console.log(`Loaded ${dataset.points.length.toLocaleString()} points`);
      console.log(`Displaying ${displayData.points.length.toLocaleString()} points`);
    } catch (error) {
      console.error('Failed to load data:', error);
      progressEl.textContent = 'Error loading data';
    }
  }

  prepareDisplayData(dataset) {
    if (dataset.points.length <= this.maxVisiblePoints) {
      return dataset;
    }

    if (this.useSampling) {
      // 적응형 샘플링
      return adaptiveSample(dataset, this.maxVisiblePoints);
    } else {
      // 균등 샘플링
      return uniformSample(dataset, this.maxVisiblePoints);
    }
  }

  async loadWithProgress(url, onProgress) {
    const response = await fetch(url);
    const reader = response.body.getReader();
    const contentLength = response.headers.get('Content-Length');
    const total = parseInt(contentLength, 10);

    let loaded = 0;
    const chunks = [];

    while (true) {
      const { done, value } = await reader.read();

      if (done) break;

      chunks.push(value);
      loaded += value.length;

      onProgress({
        loaded,
        total,
        percentage: (loaded / total) * 100
      });
    }

    const blob = new Blob(chunks);
    const text = await blob.text();

    return this.parseCSV(text);
  }

  parseCSV(text) {
    const lines = text.trim().split('\n');
    const points = lines.map(line => {
      const [x, y] = line.split(',').map(Number);
      return { x, y };
    });

    return { points };
  }

  exportData() {
    const csv = this.fullDataset.points
      .map(p => `${p.x},${p.y}`)
      .join('\n');

    const blob = new Blob([csv], { type: 'text/csv' });
    const url = URL.createObjectURL(blob);

    const a = document.createElement('a');
    a.href = url;
    a.download = 'chart-data.csv';
    a.click();

    URL.revokeObjectURL(url);
  }

  getStats() {
    return {
      totalPoints: this.fullDataset.points.length,
      displayedPoints: this.chart.getData().points.length,
      samplingRatio: (this.chart.getData().points.length /
                     this.fullDataset.points.length * 100).toFixed(2) + '%'
    };
  }
}

// 사용
const chart = new LargeDatasetChart(
  document.getElementById('chart'),
  {
    type: 'line',
    title: 'Large Dataset',
    showGrid: true
  },
  {
    maxVisiblePoints: 5000,
    useSampling: true
  }
);

// 데이터 로드
chart.loadData('/api/large-dataset.csv');

// 통계 표시
setTimeout(() => {
  console.log(chart.getStats());
}, 5000);
```
