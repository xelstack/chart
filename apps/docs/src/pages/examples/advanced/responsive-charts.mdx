# Responsive Charts

화면 크기에 따라 자동으로 조정되는 반응형 차트를 만드는 방법입니다.

## 기본 반응형 차트

```typescript
import { createChart } from '@xelstack/chart-core';

function createResponsiveChart(container, dataset, config) {
  // 컨테이너 크기에 맞춰 차트 생성
  const rect = container.getBoundingClientRect();

  const responsiveConfig = {
    ...config,
    width: rect.width,
    height: rect.height
  };

  const chart = createChart(container, dataset, responsiveConfig);

  // 리사이즈 이벤트 처리
  window.addEventListener('resize', () => {
    const newRect = container.getBoundingClientRect();

    chart.updateConfig({
      width: newRect.width,
      height: newRect.height
    });
  });

  return chart;
}

// 사용
const chart = createResponsiveChart(
  document.getElementById('chart'),
  dataset,
  { type: 'line', title: 'Responsive Chart' }
);
```

## ResizeObserver 사용

더 정확한 리사이즈 감지:

```typescript
function createResponsiveChartWithObserver(container, dataset, config) {
  const rect = container.getBoundingClientRect();

  const chart = createChart(container, dataset, {
    ...config,
    width: rect.width,
    height: rect.height
  });

  // ResizeObserver 설정
  const resizeObserver = new ResizeObserver((entries) => {
    for (const entry of entries) {
      const { width, height } = entry.contentRect;

      chart.updateConfig({ width, height });
    }
  });

  resizeObserver.observe(container);

  // 정리 함수
  return {
    chart,
    destroy: () => {
      resizeObserver.disconnect();
      chart.destroy();
    }
  };
}
```

## 디바운싱

리사이즈 이벤트 최적화:

```typescript
function debounce(fn, delay) {
  let timeoutId;
  return (...args) => {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => fn(...args), delay);
  };
}

const chart = createChart(container, dataset, config);

const handleResize = debounce(() => {
  const rect = container.getBoundingClientRect();

  chart.updateConfig({
    width: rect.width,
    height: rect.height
  });
}, 250);

window.addEventListener('resize', handleResize);
```

## 브레이크포인트 기반

모바일/태블릿/데스크톱 별 설정:

```typescript
function getResponsiveConfig(width) {
  if (width < 576) {
    // 모바일
    return {
      width: width - 32,
      height: 250,
      lineWidth: 2,
      pointRadius: 3,
      showGrid: false
    };
  } else if (width < 992) {
    // 태블릿
    return {
      width: width - 40,
      height: 350,
      lineWidth: 2,
      pointRadius: 4,
      showGrid: true
    };
  } else {
    // 데스크톱
    return {
      width: Math.min(width - 48, 1200),
      height: 500,
      lineWidth: 3,
      pointRadius: 5,
      showGrid: true
    };
  }
}

function createBreakpointChart(container, dataset, baseConfig) {
  const width = container.clientWidth;
  const responsiveConfig = {
    ...baseConfig,
    ...getResponsiveConfig(width)
  };

  const chart = createChart(container, dataset, responsiveConfig);

  window.addEventListener('resize', debounce(() => {
    const newWidth = container.clientWidth;
    const newConfig = getResponsiveConfig(newWidth);

    chart.updateConfig(newConfig);
  }, 250));

  return chart;
}
```

## 가로/세로 모드 감지

```typescript
function createOrientationAwareChart(container, dataset, config) {
  const isLandscape = window.innerWidth > window.innerHeight;

  const orientationConfig = {
    ...config,
    width: isLandscape ? 800 : 350,
    height: isLandscape ? 400 : 450
  };

  const chart = createChart(container, dataset, orientationConfig);

  window.addEventListener('orientationchange', () => {
    setTimeout(() => {
      const newIsLandscape = window.innerWidth > window.innerHeight;

      chart.updateConfig({
        width: newIsLandscape ? 800 : 350,
        height: newIsLandscape ? 400 : 450
      });
    }, 100);
  });

  return chart;
}
```

## 완전한 반응형 클래스

```typescript
class ResponsiveChart {
  constructor(container, dataset, config, options = {}) {
    this.container = container;
    this.debounceDelay = options.debounceDelay || 250;
    this.breakpoints = options.breakpoints || {
      mobile: 576,
      tablet: 992
    };

    this.chart = this.createChart(dataset, config);
    this.setupResizeHandling();
  }

  createChart(dataset, config) {
    const responsiveConfig = this.getResponsiveConfig(config);
    return createChart(this.container, dataset, responsiveConfig);
  }

  getResponsiveConfig(baseConfig) {
    const width = this.container.clientWidth;
    const device = this.getDeviceType(width);

    const deviceConfigs = {
      mobile: {
        width: width - 32,
        height: 250,
        lineWidth: 2,
        pointRadius: 3,
        showGrid: false
      },
      tablet: {
        width: width - 40,
        height: 350,
        lineWidth: 2,
        pointRadius: 4,
        showGrid: true
      },
      desktop: {
        width: Math.min(width - 48, 1200),
        height: 500,
        lineWidth: 3,
        pointRadius: 5,
        showGrid: true
      }
    };

    return {
      ...baseConfig,
      ...deviceConfigs[device]
    };
  }

  getDeviceType(width) {
    if (width < this.breakpoints.mobile) {
      return 'mobile';
    } else if (width < this.breakpoints.tablet) {
      return 'tablet';
    } else {
      return 'desktop';
    }
  }

  setupResizeHandling() {
    // ResizeObserver 사용 (더 정확함)
    if (typeof ResizeObserver !== 'undefined') {
      this.resizeObserver = new ResizeObserver(
        this.debounce(() => {
          this.handleResize();
        }, this.debounceDelay)
      );

      this.resizeObserver.observe(this.container);
    } else {
      // 폴백: window resize 이벤트
      this.resizeHandler = this.debounce(() => {
        this.handleResize();
      }, this.debounceDelay);

      window.addEventListener('resize', this.resizeHandler);
    }

    // 오리엔테이션 변경 감지
    window.addEventListener('orientationchange', () => {
      setTimeout(() => this.handleResize(), 100);
    });
  }

  handleResize() {
    const width = this.container.clientWidth;
    const device = this.getDeviceType(width);
    const currentConfig = this.chart.getConfig();

    // 디바이스 타입이 변경된 경우에만 업데이트
    const newConfig = this.getResponsiveConfig(currentConfig);

    this.chart.updateConfig(newConfig);

    console.log(`Resized to ${device}: ${width}px`);
  }

  debounce(fn, delay) {
    let timeoutId;
    return (...args) => {
      clearTimeout(timeoutId);
      timeoutId = setTimeout(() => fn.apply(this, args), delay);
    };
  }

  destroy() {
    if (this.resizeObserver) {
      this.resizeObserver.disconnect();
    }

    if (this.resizeHandler) {
      window.removeEventListener('resize', this.resizeHandler);
    }

    this.chart.destroy();
  }
}

// 사용
const responsiveChart = new ResponsiveChart(
  document.getElementById('chart'),
  dataset,
  {
    type: 'line',
    title: 'Responsive Chart'
  },
  {
    debounceDelay: 250,
    breakpoints: {
      mobile: 576,
      tablet: 992
    }
  }
);

// 정리
// responsiveChart.destroy();
```

## React 통합

```jsx
import { useEffect, useRef, useState } from 'react';
import { createChart } from '@xelstack/chart-core';

function ResponsiveChart({ data, config }) {
  const containerRef = useRef(null);
  const chartRef = useRef(null);

  useEffect(() => {
    if (!containerRef.current) return;

    const rect = containerRef.current.getBoundingClientRect();

    chartRef.current = createChart(
      containerRef.current,
      data,
      {
        ...config,
        width: rect.width,
        height: rect.height
      }
    );

    // ResizeObserver
    const observer = new ResizeObserver((entries) => {
      for (const entry of entries) {
        const { width, height } = entry.contentRect;
        chartRef.current?.updateConfig({ width, height });
      }
    });

    observer.observe(containerRef.current);

    return () => {
      observer.disconnect();
      chartRef.current?.destroy();
    };
  }, []);

  useEffect(() => {
    chartRef.current?.updateData(data);
  }, [data]);

  return (
    <div
      ref={containerRef}
      style={{ width: '100%', height: '100%' }}
    />
  );
}
```
