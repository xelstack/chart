# Simple Bar Chart

기본 막대 차트를 만드는 방법입니다.

## 기본 예제

<ChartExample
  title="Simple Bar Chart"
  description="월별 판매량을 막대 차트로 표시"
  chartType="bar"
  data={{
    points: [
      { x: 1, y: 30 },
      { x: 2, y: 45 },
      { x: 3, y: 35 },
      { x: 4, y: 50 },
      { x: 5, y: 42 }
    ]
  }}
  config={{
    width: 800,
    height: 400,
    colors: ['#3366ff'],
    showGrid: true
  }}
  code={`import { createChart } from '@xelstack/chart-core';

const dataset = {
  points: [
    { x: 1, y: 30 },
    { x: 2, y: 45 },
    { x: 3, y: 35 },
    { x: 4, y: 50 },
    { x: 5, y: 42 }
  ]
};

const config = {
  type: 'bar',
  title: 'Monthly Sales',
  width: 800,
  height: 400,
  colors: ['#3366ff'],
  showGrid: true
};

const chart = createChart(
  document.getElementById('chart'),
  dataset,
  config
);`}
  client:load
/>
