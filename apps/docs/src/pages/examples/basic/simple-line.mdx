# Simple Line Chart

가장 기본적인 선 차트를 만드는 방법입니다.

## 기본 예제

<ChartExample
  title="Simple Line Chart"
  description="월별 데이터를 선 차트로 표시"
  chartType="line"
  data={{
    points: [
      { x: 1, y: 10 },
      { x: 2, y: 25 },
      { x: 3, y: 17 },
      { x: 4, y: 32 },
      { x: 5, y: 28 }
    ]
  }}
  config={{
    width: 800,
    height: 400,
    colors: ['#3366ff'],
    showGrid: true
  }}
  code={`import { createChart } from '@xelstack/chart-core';

const dataset = {
  points: [
    { x: 1, y: 10 },
    { x: 2, y: 25 },
    { x: 3, y: 17 },
    { x: 4, y: 32 },
    { x: 5, y: 28 }
  ]
};

const config = {
  type: 'line',
  title: 'Simple Line Chart',
  width: 800,
  height: 400,
  showGrid: true
};

const chart = createChart(
  document.getElementById('chart'),
  dataset,
  config
);`}
  client:load
/>

  code={`const dataset = {

    { x: 2, y: 25, series: 'Series A' },

  type: 'line',
  colors: ['#3366ff', '#ff3366'],

};`}
  client:load
/>
