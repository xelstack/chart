# Simple Scatter Chart

기본 산점도를 만드는 방법입니다.

## 기본 예제

<ChartExample
  title="Simple Scatter Chart"
  description="데이터 포인트 간의 관계를 시각화"
  chartType="scatter"
  data={{
    points: [
      { x: 10, y: 20 },
      { x: 15, y: 35 },
      { x: 25, y: 30 },
      { x: 30, y: 45 },
      { x: 40, y: 40 },
      { x: 50, y: 55 }
    ]
  }}
  config={{
    width: 800,
    height: 400,
    colors: ['#3366ff'],
    showGrid: true
  }}
  code={`import { createChart } from '@xelstack/chart-core';

const dataset = {
  points: [
    { x: 10, y: 20 },
    { x: 15, y: 35 },
    { x: 25, y: 30 },
    { x: 30, y: 45 },
    { x: 40, y: 40 },
    { x: 50, y: 55 }
  ]
};

const config = {
  type: 'scatter',
  title: 'Scatter Plot',
  width: 800,
  height: 400,
  colors: ['#3366ff'],
  pointRadius: 5,
  showGrid: true
};

const chart = createChart(
  document.getElementById('chart'),
  dataset,
  config
);`}
  client:load
/>

## 다중 그룹

};`}

/>
