# Integration Examples (통합 예제)

다양한 프레임워크 및 라이브러리와 XelStack Chart를 통합하는 예제입니다.

## React

### 기본 React 컴포넌트

XelStack Chart를 React 컴포넌트로 래핑합니다.

```tsx
import { useEffect, useRef } from 'react';
import {
  createChart,
  type ChartConfig,
  type Dataset,
  type Chart,
} from '@xelstack/chart-core';

interface ChartProps {
  data: Dataset;
  config: ChartConfig;
}

export default function ChartComponent({ data, config }: ChartProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const chartRef = useRef<Chart | null>(null);

  // 초기 차트 생성
  useEffect(() => {
    if (containerRef.current === null) return;

    chartRef.current = createChart(containerRef.current, data, config);

    return () => {
      chartRef.current?.destroy();
    };
  }, []);

  // 데이터 변경 시 업데이트
  useEffect(() => {
    if (chartRef.current !== null) {
      chartRef.current.updateData(data);
      chartRef.current.render();
    }
  }, [data]);

  // 설정 변경 시 업데이트
  useEffect(() => {
    if (chartRef.current !== null) {
      chartRef.current.updateConfig(config);
      chartRef.current.render();
    }
  }, [config]);

  return <div ref={containerRef} className="chart-container" />;
}
```

### 사용 예제

```tsx
import ChartComponent from './ChartComponent';

function App() {
  const [data, setData] = useState({
    points: [
      { x: 'Jan', y: 100 },
      { x: 'Feb', y: 150 },
      { x: 'Mar', y: 120 },
    ],
  });

  const config = {
    type: 'line' as const,
    width: 800,
    height: 400,
    colors: ['#3366ff'],
    showGrid: true,
  };

  return (
    <div>
      <h1>My Chart</h1>
      <ChartComponent data={data} config={config} />
      <button onClick={() => setData({
        points: [
          { x: 'Jan', y: Math.random() * 200 },
          { x: 'Feb', y: Math.random() * 200 },
          { x: 'Mar', y: Math.random() * 200 },
        ],
      })}>
        Update Data
      </button>
    </div>
  );
}
```

### Custom Hook

재사용 가능한 커스텀 훅으로 만듭니다.

```tsx
import { useEffect, useRef, useState } from 'react';
import {
  createChart,
  type ChartConfig,
  type Dataset,
  type Chart,
} from '@xelstack/chart-core';

export function useChart(
  data: Dataset,
  config: ChartConfig
) {
  const containerRef = useRef<HTMLDivElement>(null);
  const chartRef = useRef<Chart | null>(null);
  const [isReady, setIsReady] = useState(false);

  // 초기화
  useEffect(() => {
    if (containerRef.current === null) return;

    chartRef.current = createChart(containerRef.current, data, config);
    setIsReady(true);

    return () => {
      chartRef.current?.destroy();
      chartRef.current = null;
      setIsReady(false);
    };
  }, []);

  // 데이터 업데이트
  useEffect(() => {
    if (isReady && chartRef.current !== null) {
      chartRef.current.updateData(data);
      chartRef.current.render();
    }
  }, [data, isReady]);

  // 설정 업데이트
  useEffect(() => {
    if (isReady && chartRef.current !== null) {
      chartRef.current.updateConfig(config);
      chartRef.current.render();
    }
  }, [config, isReady]);

  return {
    containerRef,
    chart: chartRef.current,
    isReady,
  };
}

// 사용 예제
function MyChart() {
  const { containerRef } = useChart(
    { points: [{ x: 'A', y: 100 }] },
    { type: 'line', colors: ['#3366ff'] }
  );

  return <div ref={containerRef} />;
}
```

## Vue 3

### Composition API

Vue 3 Composition API를 사용한 컴포넌트입니다.

```vue
<template>
  <div ref="chartContainer" class="chart-container"></div>
</template>

<script setup lang="ts">
import { ref, onMounted, onUnmounted, watch } from 'vue';
import {
  createChart,
  type ChartConfig,
  type Dataset,
  type Chart,
} from '@xelstack/chart-core';

interface Props {
  data: Dataset;
  config: ChartConfig;
}

const props = defineProps<Props>();
const chartContainer = ref<HTMLDivElement | null>(null);
let chartInstance: Chart | null = null;

onMounted(() => {
  if (chartContainer.value !== null) {
    chartInstance = createChart(
      chartContainer.value,
      props.data,
      props.config
    );
  }
});

onUnmounted(() => {
  chartInstance?.destroy();
});

watch(
  () => props.data,
  (newData) => {
    if (chartInstance !== null) {
      chartInstance.updateData(newData);
      chartInstance.render();
    }
  },
  { deep: true }
);

watch(
  () => props.config,
  (newConfig) => {
    if (chartInstance !== null) {
      chartInstance.updateConfig(newConfig);
      chartInstance.render();
    }
  },
  { deep: true }
);
</script>

<style scoped>
.chart-container {
  width: 100%;
  height: 400px;
}
</style>
```

### Composable

재사용 가능한 composable로 만듭니다.

```typescript
// useChart.ts
import { ref, onMounted, onUnmounted, watch, type Ref } from 'vue';
import {
  createChart,
  type ChartConfig,
  type Dataset,
  type Chart,
} from '@xelstack/chart-core';

export function useChart(
  data: Ref<Dataset>,
  config: Ref<ChartConfig>
) {
  const containerRef = ref<HTMLDivElement | null>(null);
  const chartInstance = ref<Chart | null>(null);
  const isReady = ref(false);

  onMounted(() => {
    if (containerRef.value !== null) {
      chartInstance.value = createChart(
        containerRef.value,
        data.value,
        config.value
      );
      isReady.value = true;
    }
  });

  onUnmounted(() => {
    chartInstance.value?.destroy();
  });

  watch(
    data,
    (newData) => {
      if (chartInstance.value !== null) {
        chartInstance.value.updateData(newData);
        chartInstance.value.render();
      }
    },
    { deep: true }
  );

  watch(
    config,
    (newConfig) => {
      if (chartInstance.value !== null) {
        chartInstance.value.updateConfig(newConfig);
        chartInstance.value.render();
      }
    },
    { deep: true }
  );

  return {
    containerRef,
    chart: chartInstance,
    isReady,
  };
}
```

## Svelte

### 기본 Svelte 컴포넌트

Svelte로 차트를 래핑합니다.

```svelte
<script lang="ts">
  import { onMount, onDestroy } from 'svelte';
  import {
    createChart,
    type ChartConfig,
    type Dataset,
    type Chart,
  } from '@xelstack/chart-core';

  export let data: Dataset;
  export let config: ChartConfig;

  let container: HTMLDivElement;
  let chart: Chart | null = null;

  onMount(() => {
    if (container) {
      chart = createChart(container, data, config);
    }
  });

  onDestroy(() => {
    chart?.destroy();
  });

  // 데이터 변경 시 업데이트
  $: if (chart) {
    chart.updateData(data);
    chart.render();
  }

  // 설정 변경 시 업데이트
  $: if (chart) {
    chart.updateConfig(config);
    chart.render();
  }
</script>

<div bind:this={container} class="chart-container"></div>

<style>
  .chart-container {
    width: 100%;
    height: 400px;
  }
</style>
```

### 사용 예제

## TypeScript 타입 안전성

### 제네릭 컴포넌트

타입 안전한 차트 컴포넌트를 만듭니다.

```tsx
import { useEffect, useRef } from 'react';
import {
  createChart,
  type ChartConfig,
  type Dataset,
  type Chart,
} from '@xelstack/chart-core';

interface TypedChartProps<T extends ChartConfig['type']> {
  data: Dataset;
  config: ChartConfig & { type: T };
  onReady?: (chart: Chart) => void;
}

export function TypedChart<T extends ChartConfig['type']>({
  data,
  config,
  onReady,
}: TypedChartProps<T>) {
  const containerRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (containerRef.current === null) return;

    const chart = createChart(containerRef.current, data, config);

    onReady?.(chart);

    return () => {
      chart.destroy();
    };
  }, [data, config, onReady]);

  return <div ref={containerRef} />;
}

// 사용 예제
function App() {
  return (
    <TypedChart
      data={{ points: [{ x: 'A', y: 100 }] }}
      config={{
        type: 'line', // 타입 추론됨
        colors: ['#3366ff'],
        showGrid: true,
      }}
      onReady={(chart) => {
        console.log('Chart ready:', chart);
      }}
    />
  );
}
```
