# Click Events

차트의 데이터 포인트나 영역을 클릭했을 때 이벤트를 처리하는 방법입니다.

## 기본 클릭 이벤트

```typescript
import { createChart } from '@xelstack/chart-core';

const chart = createChart(container, dataset, config);
const canvas = container.querySelector('canvas');

canvas.addEventListener('click', (event) => {
  const rect = canvas.getBoundingClientRect();
  const x = event.clientX - rect.left;
  const y = event.clientY - rect.top;

  // 클릭된 위치의 데이터 포인트 찾기
  const clickedPoint = findPointAtPosition(x, y, dataset, canvas);

  if (clickedPoint) {
    console.log('Clicked point:', clickedPoint);
    // 포인트 선택 처리
    handlePointClick(clickedPoint);
  }
});

function findPointAtPosition(x, y, dataset, canvas) {
  const CLICK_THRESHOLD = 10; // 픽셀 단위

  for (const point of dataset.points) {
    const pixelX = (point.x / 100) * canvas.width;
    const pixelY = canvas.height - (point.y / 100) * canvas.height;

    const distance = Math.hypot(x - pixelX, y - pixelY);

    if (distance <= CLICK_THRESHOLD) {
      return point;
    }
  }

  return null;
}
```

## 포인트 선택 및 하이라이트

```typescript
let selectedPoint = null;

function handlePointClick(point) {
  // 이전 선택 해제
  if (selectedPoint) {
    selectedPoint.selected = false;
  }

  // 새 포인트 선택
  selectedPoint = point;
  point.selected = true;

  // 선택된 포인트 강조 표시
  highlightPoint(point);

  // 선택 정보 표시
  displayPointInfo(point);

  // 차트 재렌더링
  chart.render();
}

function highlightPoint(point) {
  // 선택된 포인트에 특별한 스타일 적용
  const ctx = canvas.getContext('2d');

  const pixelX = (point.x / 100) * canvas.width;
  const pixelY = canvas.height - (point.y / 100) * canvas.height;

  // 하이라이트 원 그리기
  ctx.beginPath();
  ctx.arc(pixelX, pixelY, 8, 0, Math.PI * 2);
  ctx.strokeStyle = '#ff0000';
  ctx.lineWidth = 3;
  ctx.stroke();
}

function displayPointInfo(point) {
  const infoPanel = document.getElementById('point-info');
  infoPanel.innerHTML = `
    <h3>Selected Point</h3>
    <p>X: ${point.x}</p>
    <p>Y: ${point.y}</p>
    ${point.series ? `<p>Series: ${point.series}</p>` : ''}
  `;
}
```

## 더블 클릭 이벤트

```typescript
let clickTimeout = null;
const DOUBLE_CLICK_DELAY = 300;

canvas.addEventListener('click', (event) => {
  if (clickTimeout) {
    // 더블 클릭
    clearTimeout(clickTimeout);
    clickTimeout = null;
    handleDoubleClick(event);
  } else {
    // 싱글 클릭
    clickTimeout = setTimeout(() => {
      handleSingleClick(event);
      clickTimeout = null;
    }, DOUBLE_CLICK_DELAY);
  }
});
```

## 우클릭 컨텍스트 메뉴

```typescript
canvas.addEventListener('contextmenu', (event) => {
  event.preventDefault();

  const rect = canvas.getBoundingClientRect();
  const x = event.clientX - rect.left;
  const y = event.clientY - rect.top;

  const point = findPointAtPosition(x, y, dataset, canvas);

  if (point) {
    showContextMenu(event.clientX, event.clientY, point);
  }
});

function showContextMenu(x, y, point) {
  const menu = document.createElement('div');
  menu.style.cssText = `
    position: absolute;
    left: ${x}px;
    top: ${y}px;
    background: white;
    border: 1px solid #ccc;
    border-radius: 4px;
    box-shadow: 0 2px 8px rgba(0,0,0,0.15);
    z-index: 1000;
    padding: 4px 0;
  `;

  const options = [
    { label: 'View Details', action: () => viewDetails(point) },
    { label: 'Delete Point', action: () => deletePoint(point) },
    { label: 'Export Data', action: () => exportPoint(point) }
  ];

  options.forEach(option => {
    const item = document.createElement('div');
    item.textContent = option.label;
    item.style.cssText = `
      padding: 8px 16px;
      cursor: pointer;
    `;
    item.addEventListener('click', () => {
      option.action();
      menu.remove();
    });
    item.addEventListener('mouseenter', () => {
      item.style.background = '#f0f0f0';
    });
    item.addEventListener('mouseleave', () => {
      item.style.background = 'white';
    });
    menu.appendChild(item);
  });

  document.body.appendChild(menu);

  // 외부 클릭 시 메뉴 닫기
  setTimeout(() => {
    document.addEventListener('click', () => menu.remove(), { once: true });
  }, 0);
}
```

## 영역 선택 (박스 선택)

```typescript
let isSelecting = false;
let selectionStart = null;
let selectionBox = null;

canvas.addEventListener('mousedown', (event) => {
  if (event.shiftKey) {
    isSelecting = true;
    const rect = canvas.getBoundingClientRect();
    selectionStart = {
      x: event.clientX - rect.left,
      y: event.clientY - rect.top
    };

    // 선택 박스 엘리먼트 생성
    selectionBox = document.createElement('div');
    selectionBox.style.cssText = `
      position: absolute;
      border: 2px dashed #3366ff;
      background: rgba(51, 102, 255, 0.1);
      pointer-events: none;
    `;
    container.appendChild(selectionBox);
  }
});

canvas.addEventListener('mousemove', (event) => {
  if (!isSelecting) return;

  const rect = canvas.getBoundingClientRect();
  const currentX = event.clientX - rect.left;
  const currentY = event.clientY - rect.top;

  // 선택 박스 업데이트
  const left = Math.min(selectionStart.x, currentX);
  const top = Math.min(selectionStart.y, currentY);
  const width = Math.abs(currentX - selectionStart.x);
  const height = Math.abs(currentY - selectionStart.y);

  selectionBox.style.left = `${left}px`;
  selectionBox.style.top = `${top}px`;
  selectionBox.style.width = `${width}px`;
  selectionBox.style.height = `${height}px`;
});

canvas.addEventListener('mouseup', (event) => {
  if (!isSelecting) return;

  const rect = canvas.getBoundingClientRect();
  const endX = event.clientX - rect.left;
  const endY = event.clientY - rect.top;

  // 박스 안의 포인트 찾기
  const selectedPoints = findPointsInBox(
    Math.min(selectionStart.x, endX),
    Math.min(selectionStart.y, endY),
    Math.abs(endX - selectionStart.x),
    Math.abs(endY - selectionStart.y),
    dataset,
    canvas
  );

  console.log(`Selected ${selectedPoints.length} points`);
  handleMultiplePointsSelected(selectedPoints);

  // 정리
  selectionBox.remove();
  isSelecting = false;
  selectionStart = null;
});

function findPointsInBox(x, y, width, height, dataset, canvas) {
  return dataset.points.filter(point => {
    const pixelX = (point.x / 100) * canvas.width;
    const pixelY = canvas.height - (point.y / 100) * canvas.height;

    return pixelX >= x && pixelX <= x + width &&
           pixelY >= y && pixelY <= y + height;
  });
}
```

## 완전한 예제

```typescript
class InteractiveChart {
  constructor(container, dataset, config) {
    this.chart = createChart(container, dataset, config);
    this.canvas = container.querySelector('canvas');
    this.selectedPoints = new Set();
    this.setupEventListeners();
  }

  setupEventListeners() {
    // 클릭 이벤트
    this.canvas.addEventListener('click', (e) => {
      this.handleClick(e);
    });

    // 더블 클릭 이벤트
    this.canvas.addEventListener('dblclick', (e) => {
      this.handleDoubleClick(e);
    });

    // 우클릭 이벤트
    this.canvas.addEventListener('contextmenu', (e) => {
      e.preventDefault();
      this.handleRightClick(e);
    });
  }

  handleClick(event) {
    const point = this.findPointAtEvent(event);

    if (point) {
      if (event.ctrlKey || event.metaKey) {
        // Ctrl/Cmd + 클릭: 다중 선택
        this.togglePointSelection(point);
      } else {
        // 일반 클릭: 단일 선택
        this.selectSinglePoint(point);
      }

      this.updateChart();
    } else {
      // 빈 공간 클릭: 모든 선택 해제
      this.clearSelection();
      this.updateChart();
    }
  }

  handleDoubleClick(event) {
    const point = this.findPointAtEvent(event);

    if (point) {
      // 포인트 편집 다이얼로그 표시
      this.showEditDialog(point);
    }
  }

  handleRightClick(event) {
    const point = this.findPointAtEvent(event);

    if (point) {
      this.showContextMenu(event, point);
    }
  }

  findPointAtEvent(event) {
    const rect = this.canvas.getBoundingClientRect();
    const x = event.clientX - rect.left;
    const y = event.clientY - rect.top;

    const data = this.chart.getData().points;

    for (const point of data) {
      const pixelX = this.dataToPixelX(point.x);
      const pixelY = this.dataToPixelY(point.y);
      const distance = Math.hypot(x - pixelX, y - pixelY);

      if (distance <= 10) {
        return point;
      }
    }

    return null;
  }

  togglePointSelection(point) {
    if (this.selectedPoints.has(point)) {
      this.selectedPoints.delete(point);
    } else {
      this.selectedPoints.add(point);
    }
  }

  selectSinglePoint(point) {
    this.selectedPoints.clear();
    this.selectedPoints.add(point);
  }

  clearSelection() {
    this.selectedPoints.clear();
  }

  updateChart() {
    // 선택된 포인트 하이라이트
    this.highlightSelectedPoints();
    this.chart.render();
  }

  highlightSelectedPoints() {
    const ctx = this.canvas.getContext('2d');

    this.selectedPoints.forEach(point => {
      const x = this.dataToPixelX(point.x);
      const y = this.dataToPixelY(point.y);

      ctx.beginPath();
      ctx.arc(x, y, 8, 0, Math.PI * 2);
      ctx.strokeStyle = '#ff0000';
      ctx.lineWidth = 2;
      ctx.stroke();
    });
  }

  showEditDialog(point) {
    const newValue = prompt(`Edit Y value for X=${point.x}`, point.y);

    if (newValue !== null) {
      point.y = parseFloat(newValue);
      this.chart.updateData(this.chart.getData());
    }
  }

  showContextMenu(event, point) {
    // 컨텍스트 메뉴 구현 (위 예제 참조)
  }

  dataToPixelX(x) {
    // 데이터 좌표를 픽셀로 변환
    return (x / 100) * this.canvas.width;
  }

  dataToPixelY(y) {
    // 데이터 좌표를 픽셀로 변환
    return this.canvas.height - (y / 100) * this.canvas.height;
  }
}

// 사용
const interactiveChart = new InteractiveChart(
  document.getElementById('chart'),
  dataset,
  { type: 'scatter', title: 'Interactive Chart' }
);
```
