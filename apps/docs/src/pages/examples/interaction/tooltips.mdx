# Tooltips

마우스 호버 시 데이터 포인트 정보를 툴팁으로 표시하는 방법입니다.

## 기본 툴팁 구현

```typescript
import { createChart } from '@xelstack/chart-core';

const chart = createChart(container, dataset, config);
const canvas = container.querySelector('canvas');

// 툴팁 엘리먼트 생성
const tooltip = document.createElement('div');
tooltip.style.position = 'absolute';
tooltip.style.background = 'rgba(0, 0, 0, 0.8)';
tooltip.style.color = 'white';
tooltip.style.padding = '8px 12px';
tooltip.style.borderRadius = '4px';
tooltip.style.fontSize = '14px';
tooltip.style.pointerEvents = 'none';
tooltip.style.display = 'none';
tooltip.style.zIndex = '1000';
document.body.appendChild(tooltip);

// 마우스 이동 이벤트
canvas.addEventListener('mousemove', (event) => {
  const rect = canvas.getBoundingClientRect();
  const x = event.clientX - rect.left;
  const y = event.clientY - rect.top;

  // 가장 가까운 데이터 포인트 찾기
  const nearestPoint = findNearestPoint(x, y, dataset, canvas);

  if (nearestPoint && nearestPoint.distance < 20) {
    // 툴팁 표시
    tooltip.style.display = 'block';
    tooltip.style.left = `${event.clientX + 10}px`;
    tooltip.style.top = `${event.clientY + 10}px`;
    tooltip.innerHTML = `
      <strong>X:</strong> ${nearestPoint.point.x}<br>
      <strong>Y:</strong> ${nearestPoint.point.y}
    `;
  } else {
    tooltip.style.display = 'none';
  }
});

canvas.addEventListener('mouseleave', () => {
  tooltip.style.display = 'none';
});

// 가장 가까운 포인트 찾기
function findNearestPoint(x, y, dataset, canvas) {
  const data = dataset.points;
  let nearest = null;
  let minDistance = Infinity;

  data.forEach((point) => {
    // 데이터 좌표를 픽셀 좌표로 변환
    const pixelX = (point.x / 100) * canvas.width;
    const pixelY = canvas.height - (point.y / 100) * canvas.height;

    const distance = Math.hypot(x - pixelX, y - pixelY);

    if (distance < minDistance) {
      minDistance = distance;
      nearest = { point, distance };
    }
  });

  return nearest;
}
```

## 고급 툴팁

커스텀 스타일과 포맷팅:

```typescript
function createTooltip() {
  const tooltip = document.createElement('div');
  tooltip.className = 'chart-tooltip';
  tooltip.style.cssText = `
    position: absolute;
    background: linear-gradient(135deg, #667eea 0%, #764ba2 100%);
    color: white;
    padding: 12px 16px;
    border-radius: 8px;
    font-size: 13px;
    box-shadow: 0 4px 6px rgba(0, 0, 0, 0.1);
    pointer-events: none;
    display: none;
    z-index: 1000;
    transition: opacity 0.2s;
  `;
  document.body.appendChild(tooltip);
  return tooltip;
}

function formatTooltipContent(point) {
  return `
    <div style="margin-bottom: 4px; font-weight: bold;">
      ${point.series || 'Data Point'}
    </div>
    <div style="display: flex; justify-content: space-between; gap: 12px;">
      <span>X: ${point.x}</span>
      <span>Y: ${point.y.toFixed(2)}</span>
    </div>
  `;
}
```

## 다중 시리즈 툴팁

여러 시리즈의 데이터를 한 번에 표시:

```typescript
function showMultiSeriesTooltip(x, dataset) {
  const pointsAtX = dataset.points.filter(p => p.x === x);

  if (pointsAtX.length === 0) return;

  const html = `
    <div style="font-weight: bold; margin-bottom: 8px;">
      X: ${x}
    </div>
    ${pointsAtX.map(p => `
      <div style="display: flex; align-items: center; gap: 8px; margin: 4px 0;">
        <div style="width: 12px; height: 12px; background: ${p.color}; border-radius: 2px;"></div>
        <span>${p.series}: ${p.y.toFixed(2)}</span>
      </div>
    `).join('')}
  `;

  tooltip.innerHTML = html;
  tooltip.style.display = 'block';
}
```

## 툴팁 위치 자동 조정

화면 밖으로 나가지 않도록:

```typescript
function positionTooltip(tooltip, event) {
  const tooltipRect = tooltip.getBoundingClientRect();
  const windowWidth = window.innerWidth;
  const windowHeight = window.innerHeight;

  let left = event.clientX + 10;
  let top = event.clientY + 10;

  // 오른쪽 경계 체크
  if (left + tooltipRect.width > windowWidth) {
    left = event.clientX - tooltipRect.width - 10;
  }

  // 아래쪽 경계 체크
  if (top + tooltipRect.height > windowHeight) {
    top = event.clientY - tooltipRect.height - 10;
  }

  tooltip.style.left = `${left}px`;
  tooltip.style.top = `${top}px`;
}
```

## 완전한 예제

```typescript
class TooltipChart {
  constructor(container, dataset, config) {
    this.chart = createChart(container, dataset, config);
    this.canvas = container.querySelector('canvas');
    this.tooltip = this.createTooltip();
    this.setupEventListeners();
  }

  createTooltip() {
    const tooltip = document.createElement('div');
    tooltip.style.cssText = `
      position: absolute;
      background: rgba(0, 0, 0, 0.9);
      color: white;
      padding: 10px 14px;
      border-radius: 6px;
      font-size: 13px;
      pointer-events: none;
      display: none;
      z-index: 1000;
      box-shadow: 0 2px 8px rgba(0, 0, 0, 0.2);
    `;
    document.body.appendChild(tooltip);
    return tooltip;
  }

  setupEventListeners() {
    this.canvas.addEventListener('mousemove', (e) => {
      this.handleMouseMove(e);
    });

    this.canvas.addEventListener('mouseleave', () => {
      this.tooltip.style.display = 'none';
    });
  }

  handleMouseMove(event) {
    const rect = this.canvas.getBoundingClientRect();
    const x = event.clientX - rect.left;
    const y = event.clientY - rect.top;

    const point = this.findNearestPoint(x, y);

    if (point && point.distance < 20) {
      this.showTooltip(event, point.data);
    } else {
      this.tooltip.style.display = 'none';
    }
  }

  findNearestPoint(x, y) {
    const data = this.chart.getData().points;
    let nearest = null;
    let minDistance = Infinity;

    data.forEach((point) => {
      const pixelX = this.dataToPixelX(point.x);
      const pixelY = this.dataToPixelY(point.y);
      const distance = Math.hypot(x - pixelX, y - pixelY);

      if (distance < minDistance) {
        minDistance = distance;
        nearest = { data: point, distance };
      }
    });

    return nearest;
  }

  showTooltip(event, point) {
    this.tooltip.innerHTML = `
      <div style="margin-bottom: 4px; font-weight: 600;">
        ${point.series || 'Value'}
      </div>
      <div>X: ${point.x}</div>
      <div>Y: ${point.y.toFixed(2)}</div>
    `;

    this.tooltip.style.display = 'block';
    this.positionTooltip(event);
  }

  positionTooltip(event) {
    const rect = this.tooltip.getBoundingClientRect();
    let left = event.clientX + 10;
    let top = event.clientY + 10;

    if (left + rect.width > window.innerWidth) {
      left = event.clientX - rect.width - 10;
    }

    if (top + rect.height > window.innerHeight) {
      top = event.clientY - rect.height - 10;
    }

    this.tooltip.style.left = `${left}px`;
    this.tooltip.style.top = `${top}px`;
  }

  dataToPixelX(x) {
    // 데이터 좌표를 픽셀로 변환 (실제 구현 필요)
    return (x / 100) * this.canvas.width;
  }

  dataToPixelY(y) {
    // 데이터 좌표를 픽셀로 변환 (실제 구현 필요)
    return this.canvas.height - (y / 100) * this.canvas.height;
  }

  destroy() {
    this.tooltip.remove();
    this.chart.destroy();
  }
}

// 사용
const tooltipChart = new TooltipChart(
  document.getElementById('chart'),
  dataset,
  { type: 'line', title: 'Chart with Tooltips' }
);
```
